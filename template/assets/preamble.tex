%%  Boxed theorems

\declaretheoremstyle[
        headfont=\bfseries\sffamily\color{green_ref!70!black}, bodyfont=\normalfont,
        mdframed={
            linewidth=2pt,
            rightline=false, topline=false, bottomline=false,
            linecolor=green_ref, backgroundcolor=green_ref!5,
        }
    ]{thmgreenbox}
    
\declaretheoremstyle[
        headfont=\bfseries\sffamily\color{red_ref!70!black}, bodyfont=\normalfont,
        mdframed={
            linewidth=2pt,
            rightline=false, topline=false, bottomline=false,
            linecolor=red_ref, backgroundcolor=red_ref!5,
        }
    ]{thmredbox}
    
\declaretheoremstyle[
        headfont=\bfseries\sffamily\color{blue_ref!70!black}, bodyfont=\normalfont,

        }
    ]{thmbluebox}
